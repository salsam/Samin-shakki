1   package chess.domain.board;
2   
3   import chess.logic.movementlogic.MovementLogic;
4   import chess.domain.pieces.King;
5   import chess.domain.pieces.Piece;
6   import java.util.ArrayList;
7   import java.util.HashMap;
8   import java.util.HashSet;
9   import java.util.List;
10  import java.util.Map;
11  import java.util.Set;
12  
13  /**
14   * This class is responsible for keeping track of the current situation on
15   * board. This class also offers methods to access every piece on board and
16   * Squares they threaten or can move to.
17   *
18   * @author sami
19   */
20  public class ChessBoard {
21  
22      /**
23       * Table containing all the squares on the game table
24       */
25      private Square[][] table;
26      /**
27       * List of all pieces that white owns.
28       */
29      private List<Piece> whitePieces;
30      /**
31       * List of all pieces that black owns.
32       */
33      private List<Piece> blackPieces;
34      /**
35       * Set containing all squares that black threatens.
36       */
37      private Set<Square> squaresThreatenedByBlack;
38      /**
39       * Set containing all squares that white threatens.
40       */
41      private Set<Square> squaresThreatenedByWhite;
42      /**
43       * MovementLogic used to see which moves are legal on this ChessBoard.
44       */
45      private MovementLogic movementLogic;
46      /**
47       * Map containing positions of both kings.
48       */
49      private Map<Player, King> kings;
50  
51      /**
52       * Creates a new chessboard with given movement logic.
53       *
54       * @param movementLogic movement logic to be applied on this chessboard.
55       */
56      public ChessBoard(MovementLogic movementLogic) {
57          initializeBoard();
58          this.movementLogic = movementLogic;
59          this.blackPieces = new ArrayList<>();
60          this.whitePieces = new ArrayList<>();
61          this.squaresThreatenedByBlack = new HashSet();
62          this.squaresThreatenedByWhite = new HashSet();
63          this.kings = new HashMap();
64      }
65  
66      /**
67       * Initializes a new 8x8 board.
68       */
69      private void initializeBoard() {
70          table = new Square[8][8];
71          for (int i = 0; i < 8; i++) {
72              for (int j = 0; j < 8; j++) {
73                  this.table[i][j] = new Square(i, j);
74              }
75          }
76      }
77  
78      /**
79       * Returns the Square[][] saved to field board.
80       *
81       * @return Reference to Square[][] that is saved to field board.
82       */
83      public Square[][] getTable() {
84          return table;
85      }
86  
87      public MovementLogic getMovementLogic() {
88          return movementLogic;
89      }
90  
91      /**
92       * Sets the board given as parameter to field board.
93       *
94       * @param newBoard Square[][] to be saved to field board.
95       */
96      public void setTable(Square[][] newBoard) {
97          this.table = newBoard;
98      }
99  
100     /**
101      * Returns a map with references from each player to their king.
102      *
103      * @return map with references from each player to their king.
104      */
105     public Map<Player, King> getKings() {
106         return this.kings;
107     }
108 
109     /**
110      * Updates the Squares that player's pieces threaten to corresponding field.
111      * This methods uses the MovementLogic given to it in constructor to check
112      * which squares are currently threatened by player.
113      *
114      * @param player Player whose corresponding field you want to update.
115      */
116     public void updateThreatenedSquares(Player player) {
117         if (player == Player.WHITE) {
118             squaresThreatenedByWhite = movementLogic.squaresThreatenedByPlayer(Player.WHITE, this);
119         } else {
120             squaresThreatenedByBlack = movementLogic.squaresThreatenedByPlayer(Player.BLACK, this);
121         }
122     }
123 
124     /**
125      * Returns a list containing all pieces currently on board and owned by the
126      * player.
127      *
128      * @param player player whose pieces you want.
129      * @return list containing all pieces owned by the player.
130      */
131     public List<Piece> getPieces(Player player) {
132         if (player == Player.WHITE) {
133             return whitePieces;
134         } else {
135             return blackPieces;
136         }
137     }
138 
139     public void setBlackPieces(List<Piece> blackPieces) {
140         this.blackPieces = blackPieces;
141     }
142 
143     public void setWhitePieces(List<Piece> whitePieces) {
144         this.whitePieces = whitePieces;
145     }
146 
147     /**
148      * Returns the Square at given location on board.
149      *
150      * @param column Column of the Square you want.
151      * @param row Row of the Square you want.
152      * @return Square at given location.
153      */
154     public Square getSquare(int column, int row) {
155         return table[column][row];
156     }
157 
158     /**
159      * Checks if the given location is on the chessboard.
160      *
161      * @param column Column of the Square being checked.
162      * @param row Row of the Square being checked.
163      * @return true if given coordinates are within the table.
164      */
165     public boolean withinTable(int column, int row) {
166         if (column < 0 || column >= table.length) {
167             return false;
168         }
169         return !(row < 0 || row >= table[0].length);
170     }
171 
172     /**
173      * Returns a set containing all Squares that player threatens.
174      *
175      * @param player Player
176      * @return set containing all squares player threatens
177      */
178     public Set<Square> threatenedSquares(Player player) {
179         if (player == Player.WHITE) {
180             return squaresThreatenedByWhite;
181         } else {
182             return squaresThreatenedByBlack;
183         }
184     }
185 }
