3   import static chess.io.ImageLoader.getImage;
4   import chess.logic.board.Player;

12  import java.awt.Graphics;
13  import java.awt.Image;
14  import java.util.HashMap;
15  import java.util.Map;
16  
17  /**
18   *
19   * @author samisalo
20   */
21  public class PieceDrawer {
22  
23      private Map<Pair, Image> images;
24  
25      public PieceDrawer() {
26          images = new HashMap();
27          addImages();
28      }
29  
30      private void addImages() {
31          images.put(new Pair(Bishop.class, Player.BLACK), getImage("blackBishop1.png"));
32          images.put(new Pair(Bishop.class, Player.WHITE), getImage("whiteBishop1.png"));
33          images.put(new Pair(King.class, Player.BLACK), getImage("blackKing1.png"));
34          images.put(new Pair(King.class, Player.WHITE), getImage("whiteKing1.png"));
35          images.put(new Pair(Knight.class, Player.BLACK), getImage("blackKnight1.png"));
36          images.put(new Pair(Knight.class, Player.WHITE), getImage("whiteKnight1.png"));
37          images.put(new Pair(Pawn.class, Player.BLACK), getImage("blackPawn1.png"));
38          images.put(new Pair(Pawn.class, Player.WHITE), getImage("whitePawn1.png"));
39          images.put(new Pair(Queen.class, Player.BLACK), getImage("blackQueen1.png"));
40          images.put(new Pair(Queen.class, Player.WHITE), getImage("whiteQueen1.png"));
41          images.put(new Pair(Rook.class, Player.BLACK), getImage("blackRook1.png"));
42          images.put(new Pair(Rook.class, Player.WHITE), getImage("whiteRook1.png"));
43      }
44  
45      /**
46       * Draws a picture of chosen piece. Picture depends on the class of chosen
47       * piece as well as the owner of piece.
48       *
49       * @param piece piece to be drawn.
50       * @param graphics Graphics object used to draw the image.
51       * @param sideLength length of each square's sides.
52       */
53      public void draw(Piece piece, Graphics graphics, int sideLength) {
54          int x = piece.getColumn() * sideLength;
55          int y = piece.getRow() * sideLength;
56          graphics.drawImage(images.get(new Pair(piece.getClass(), piece.getOwner())), x, y, sideLength, sideLength, null);
57      }
58  }
