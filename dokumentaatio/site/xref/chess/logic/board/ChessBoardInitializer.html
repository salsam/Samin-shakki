1   package chess.logic.board;
2   
3   import chess.logic.pieces.King;
4   import chess.logic.pieces.Piece;
5   
6   /**
7    * All classes that inherit this abstract class are used to initialise different
8    * starting situation on chessboard like empty board or standard starting
9    * positions.
10   *
11   * @author samisalo
12   */
13  public abstract class ChessBoardInitializer {
14  
15      /**
16       * Adds the piece on target square to list of pieces its owner owns. Also
17       * adds a reference to Map Kings if the piece is of King class.
18       *
19       * @param target Square
20       * @param chessBoardLogic ChessBoardLogic to which piece will be added
21       */
22      public static void addPieceToOwner(Square target, ChessBoardLogic chessBoardLogic) {
23          if (target.containsAPiece()) {
24              Piece piece = target.getPiece();
25              if (piece.getClass() == King.class) {
26                  chessBoardLogic.getKings().put(piece.getOwner(), (King) piece);
27              }
28              chessBoardLogic.getPieces(piece.getOwner()).add(piece);
29          }
30      }
31  
32      public abstract void initialise(ChessBoardLogic board);
33  
34      /**
35       * Removes target piece from its owner's owned pieces list.
36       *
37       * @param piece The piece you want to remove.
38       * @param chessBoardLogic ChessBoardLogic where piece will be removed from
39       */
40      public static void removePieceFromOwner(Piece piece, ChessBoardLogic chessBoardLogic) {
41          chessBoardLogic.getPieces(piece.getOwner()).remove(piece);
42      }
43  
44      protected void clearBoard(ChessBoardLogic board) {
45          for (int i = 0; i < 8; i++) {
46              for (int j = 0; j < 8; j++) {
47                  board.getSquare(i, j).setPiece(null);
48              }
49          }
50      }
51  
52      /**
53       * Makes the square piece is located on refer to piece thus adding piece on
54       * chessboard.
55       *
56       * @param board board Piece will be placed on.
57       * @param piece piece Piece to be placed.
58       */
59      public void putPieceOnBoard(ChessBoardLogic board, Piece piece) {
60          if (board.withinTable(piece.getLocation().getcolumn(), piece.getLocation().getrow())) {
61  
62              board.getSquare(piece.getLocation().getcolumn(), piece.getLocation().getrow()).setPiece(piece);
63              addPieceToOwner(piece.getLocation(), board);
64          }
65      }
66  }
