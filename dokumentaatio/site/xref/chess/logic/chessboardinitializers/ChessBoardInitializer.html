1   package chess.logic.chessboardinitializers;
2   
3   import chess.domain.board.ChessBoard;
4   import chess.domain.board.Square;
5   import chess.domain.pieces.King;
6   import chess.domain.pieces.Piece;
7   
8   /**
9    * All classes that inherit this abstract class are used to initialize different
10   * starting situations on chessboard like empty board or standard starting
11   * positions.
12   *
13   * @author sami
14   */
15  public abstract class ChessBoardInitializer {
16  
17      /**
18       * This method will be used to initialize chessboard meaning setting up
19       * starting positions depending on which ChessBoardInitializer is used..
20       *
21       * @param board board to initialized
22       */
23      public abstract void initialize(ChessBoard board);
24  
25      /**
26       * Adds the piece on target square to list of pieces its owner owns. Also
27       * adds a reference to Map Kings if the piece is of King class.
28       *
29       * @param target Square
30       * @param chessBoard ChessBoard to which piece will be added
31       */
32      public static void addPieceToOwner(Square target, ChessBoard chessBoard) {
33          if (target.containsAPiece()) {
34              Piece piece = target.getPiece();
35              if (piece.getClass() == King.class) {
36                  chessBoard.getKings().put(piece.getOwner(), (King) piece);
37              }
38              chessBoard.getPieces(piece.getOwner()).add(piece);
39          }
40      }
41  
42      /**
43       * Removes target piece from its owner's owned pieces list.
44       *
45       * @param piece The piece you want to remove.
46       * @param chessBoard ChessBoard where piece will be removed from
47       */
48      public static void removePieceFromOwner(Piece piece, ChessBoard chessBoard) {
49          chessBoard.getPieces(piece.getOwner()).remove(piece);
50      }
51  
52      /**
53       * Marks chosen piece on chosen board as taken so player it no longer
54       * affects game.
55       *
56       * @param piece Piece to be marked as taken
57       * @param chessBoard ChessBoard on which piece is situated
58       */
59      public static void setPieceAsTaken(Piece piece, ChessBoard chessBoard) {
60          for (Piece owned : chessBoard.getPieces(piece.getOwner())) {
61              if (owned.equals(piece)) {
62                  owned.setTaken(true);
63              }
64          }
65      }
66  
67      protected void clearBoard(ChessBoard board) {
68          for (int i = 0; i < board.getTable().length; i++) {
69              for (int j = 0; j < board.getTable()[0].length; j++) {
70                  if (board.getSquare(i, j).containsAPiece()) {
71                      removePieceFromOwner(board.getSquare(i, j).getPiece(), board);
72                      board.getSquare(i, j).setPiece(null);
73                  }
74              }
75          }
76      }
77  
78      /**
79       * Makes the square piece is located on refer to piece thus adding piece on
80       * chessboard.
81       *
82       * @param board board Piece will be placed on.
83       * @param piece piece Piece to be placed.
84       */
85      public static void putPieceOnBoard(ChessBoard board, Piece piece) {
86          if (board.withinTable(piece.getColumn(), piece.getRow())) {
87  
88              board.getSquare(piece.getColumn(), piece.getRow()).setPiece(piece);
89              addPieceToOwner(board.getSquare(piece.getColumn(), piece.getRow()), board);
90          }
91      }
92  }
