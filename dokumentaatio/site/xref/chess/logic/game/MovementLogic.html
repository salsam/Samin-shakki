1   package chess.logic.game;
2   
3   import chess.logic.board.ChessBoard;
4   import chess.logic.board.Square;
5   import chess.logic.piecemovers.*;
6   import chess.logic.pieces.*;
7   import java.util.HashSet;
8   import java.util.Set;
9   
10  /**
11   * This class is responsible for movement related logic. Class offers methods to
12   * get possible moves and threatened squares of a Piece on ChessBoard given as
13   * parameter. Also offers method to move piece to target location on chessboard.
14   *
15   * @author sami
16   */
17  public class MovementLogic {
18  
19      private BishopMover bishopMover;
20      private KingMover kingMover;
21      private KnightMover knightMover;
22      private PawnMover pawnMover;
23      private QueenMover queenMover;
24      private RookMover rookMover;
25  
26      /**
27       * Creates a new MovementLogic initializing mover-objects for each chess
28       * piece-class.
29       */
30      public MovementLogic() {
31          bishopMover = new BishopMover();
32          kingMover = new KingMover();
33          knightMover = new KnightMover();
34          pawnMover = new PawnMover();
35          queenMover = new QueenMover();
36          rookMover = new RookMover();
37      }
38  
39      /**
40       * Uses corresponding mover to returns a set containing all squares given
41       * piece threatens on given board.
42       *
43       * @param piece piece of which threatened squares are being checked
44       * @param board board on which piece is placed
45       * @return a set containing all squares given piece threatens on given board
46       */
47      public Set<Square> threatenedSquares(Piece piece, ChessBoard board) {
48          if (piece.getClass() == Bishop.class) {
49              return bishopMover.threatenedSquares(piece, board);
50          } else if (piece.getClass() == King.class) {
51              return kingMover.threatenedSquares(piece, board);
52          } else if (piece.getClass() == Knight.class) {
53              return knightMover.threatenedSquares(piece, board);
54          } else if (piece.getClass() == Pawn.class) {
55              return pawnMover.threatenedSquares(piece, board);
56          } else if (piece.getClass() == Queen.class) {
57              return queenMover.threatenedSquares(piece, board);
58          } else if (piece.getClass() == Rook.class) {
59              return rookMover.threatenedSquares(piece, board);
60          }
61          return new HashSet<>();
62      }
63  
64      /**
65       * Uses corresponding mover to returns a set containing all squares given
66       * piece can move to on given board.
67       *
68       * @param piece piece of which possible moves are being checked
69       * @param board board on which piece is placed
70       * @return a set containing all squares given piece can move to on given
71       * board
72       */
73      public Set<Square> possibleMoves(Piece piece, ChessBoard board) {
74          if (piece.getClass() == Bishop.class) {
75              return bishopMover.possibleMoves(piece, board);
76          } else if (piece.getClass() == King.class) {
77              return kingMover.possibleMoves(piece, board);
78          } else if (piece.getClass() == Knight.class) {
79              return knightMover.possibleMoves(piece, board);
80          } else if (piece.getClass() == Pawn.class) {
81              return pawnMover.possibleMoves(piece, board);
82          } else if (piece.getClass() == Queen.class) {
83              return queenMover.possibleMoves(piece, board);
84          } else if (piece.getClass() == Rook.class) {
85              return rookMover.possibleMoves(piece, board);
86          }
87          return new HashSet<>();
88      }
89  
90      /**
91       * Uses corresponding PieceMover to move given piece to target square on
92       * given board.
93       *
94       * @param piece piece to be moved
95       * @param target square where piece will be moved to
96       * @param board board on which movement happens
97       */
98      public void move(Piece piece, Square target, ChessBoard board) {
99          if (piece.getClass() == Bishop.class) {
100             bishopMover.move(piece, target, board);
101         } else if (piece.getClass() == King.class) {
102             kingMover.move(piece, target, board);
103         } else if (piece.getClass() == Knight.class) {
104             knightMover.move(piece, target, board);
105         } else if (piece.getClass() == Pawn.class) {
106             pawnMover.move(piece, target, board);
107         } else if (piece.getClass() == Queen.class) {
108             queenMover.move(piece, target, board);
109         } else if (piece.getClass() == Rook.class) {
110             rookMover.move(piece, target, board);
111         }
112     }
113 
114 }
