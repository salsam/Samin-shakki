6   package chess.logic.movementlogic.piecemovers;
7   
8   import chess.domain.board.ChessBoard;
9   import java.util.Set;
10  import chess.domain.board.Square;
11  import chess.domain.pieces.Bishop;
12  import chess.domain.pieces.Piece;
13  import java.util.HashSet;
14  
15  /**
16   * This class is responsible of movement calculation of bishops.
17   *
18   * @author sami
19   */
20  public class BishopMover extends PieceMover {
21  
22      /**
23       * Creates a new BishopMover-object that contains bishop's movement related
24       * logic.
25       */
26      public BishopMover() {
27      }
28  
29      /**
30       * Return a list containing all squares that this bishop threatens.
31       *
32       * @param bishop chosen bishop
33       * @param board board where this bishop moves
34       * @return list containing all squares this bishop threatens
35       */
36      @Override
37      public Set<Square> threatenedSquares(Piece bishop, ChessBoard board) {
38          bishop = (Bishop) bishop;
39          Set<Square> possibilities = new HashSet<>();
40          addDiagonalPossibilities(board.getSquare(bishop.getColumn(), bishop.getRow()), board, possibilities);
41  
42          return possibilities;
43      }
44  
45  }
