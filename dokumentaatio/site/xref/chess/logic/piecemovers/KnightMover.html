6   package chess.logic.piecemovers;
7   
8   import chess.logic.board.ChessBoard;
9   import java.util.Set;
10  import chess.logic.board.Square;
11  import chess.logic.pieces.Piece;
12  
13  /**
14   * This class is responsible for all knight-related movement logic.
15   *
16   * @author sami
17   */
18  public class KnightMover extends PieceMover {
19  
20      /**
21       * Creates a new KnightMover-object.
22       */
23      public KnightMover() {
24      }
25  
26      /**
27       * Return a list containing all squares that given knight threatens.
28       *
29       * @param piece knight
30       * @param board board where this knight moves
31       * @return list containing all squares given knight threatens
32       */
33      @Override
34      public Set<Square> threatenedSquares(Piece piece, ChessBoard board) {
35          int[] rowChange = new int[]{-2, -2, -1, -1, 1, 1, 2, 2};
36          int[] columnChange = new int[]{1, -1, 2, -2, 2, -2, 1, -1};
37  
38          return possibilities(board.getSquare(piece.getColumn(), piece.getRow()), columnChange, rowChange, board);
39      }
40  }
