6   package chess.logic.piecemovers;
7   
8   import chess.logic.board.ChessBoard;
9   import java.util.Set;
10  import chess.logic.board.Square;
11  import chess.logic.pieces.Piece;
12  import chess.logic.pieces.Rook;
13  import java.util.HashSet;
14  
15  /**
16   * This class is responsible for all rook-related movement logic.
17   *
18   * @author sami
19   */
20  public class RookMover extends PieceMover {
21  
22      /**
23       * Creates a new RookMover-object.
24       */
25      public RookMover() {
26      }
27  
28      /**
29       * This method moves rook on the board and saves true to field hasBeenMoved.
30       *
31       * @param target square this rook is moving to.
32       * @param board board on which movement happens.
33       */
34      @Override
35      public void move(Piece piece, Square target, ChessBoard board) {
36          Rook rook = (Rook) piece;
37          rook.setHasBeenMoved(true);
38          super.move(rook, target, board);
39      }
40  
41      /**
42       * Return a list containing all squares that this rook threatens.
43       *
44       * @param board board on which this rook moves
45       * @return list containing all squares this rook threatens
46       */
47      @Override
48      public Set<Square> threatenedSquares(Piece piece, ChessBoard board) {
49          Rook rook = (Rook) piece;
50          Set<Square> possibilities = new HashSet<>();
51          addHorizontalPossibilities(board.getSquare(rook.getColumn(), rook.getRow()), board, possibilities);
52          addVerticalPossibilities(board.getSquare(rook.getColumn(), rook.getRow()), board, possibilities);
53  
54          return possibilities;
55      }
56  }
