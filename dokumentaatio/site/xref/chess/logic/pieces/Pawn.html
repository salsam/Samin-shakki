5   import java.util.HashSet;

10  public class Pawn extends Piece {
11  
12      private boolean movedTwoSquaresLastTurn;
13  
14      public Pawn(Square square, Player owner) {
15          super(square, owner);
16          movedTwoSquaresLastTurn = false;
17  
18          if (owner == Player.BLACK) {
19              this.picture = ImageLoader.getImage("blackPawn1.png");
20          } else {
21              this.picture = ImageLoader.getImage("whitePawn1.png");
22          }
23      }
24  
25      /**
26       * Returns a field to field copy of this piece.
27       *
28       * @param location location where the clone will be placed
29       * @return deep copy of this pawn
30       */
31      @Override
32      public Piece clone(Square location) {
33          return new Pawn(location, this.owner);
34      }
35  
36      @Override
37      public void move(Square target, ChessBoardLogic board) {
38          if (Math.abs(this.location.getrow() - target.getrow()) == 2) {
39              movedTwoSquaresLastTurn = true;
40          }
41          super.move(target, board);
42      }
43  
44      public boolean getMovedTwoSquaresLastTurn() {
45          return movedTwoSquaresLastTurn;
46      }
47  
48      public void setMovedTwoSquaresLastTurn(boolean movedTwoSquaresLastTurn) {
49          this.movedTwoSquaresLastTurn = movedTwoSquaresLastTurn;
50      }
51  
52      /**
53       * Return a list containing all squares that this pawn threatens. En passant
54       * is a special move in chess which only pawns can perform. It means that if
55       * opposing pawn moves two squares to be next to your own pawn, on your next
56       * turn your pawn can take it as if it had only moved one square.
57       *
58       * @param board board on which this pawn moves
59       * @return list containing all squares this pawn threatens
60       */
61      @Override
62      public Set<Square> threatenedSquares(ChessBoardLogic board) {
63          Set<Square> squares = new HashSet();
64          int[] columnChange = new int[]{1, -1};
65          int column = this.location.getcolumn();
66          int row = this.location.getrow() + this.owner.getDirection();
67          Square target;
68  
69          for (int i = 0; i < 2; i++) {
70              if (board.withinTable(column + columnChange[i], row)) {
71                  target = board.getSquare(column + columnChange[i], row);
72                  squares.add(target);
73              }
74              enPassant(board, column, columnChange, i, squares);
75          }
76  
77          return squares;
78      }
79  
80      private void enPassant(ChessBoardLogic board, int column, int[] columnChange, int i, Set<Square> squares) {
81          Square target;
82          if (board.withinTable(column + columnChange[i], this.location.getrow())) {
83              target = board.getSquare(column + columnChange[i], this.location.getrow());
84  
85              if (targetContainsAnEnemyPawn(target)) {
86                  Pawn opposingPawn = (Pawn) target.getPiece();
87                  if (opposingPawn.getMovedTwoSquaresLastTurn()) {
88                      squares.add(target);
89                  }
90              }
91          }
92      }
93  
94      private boolean targetContainsAnEnemyPawn(Square target) {
95          if (!target.containsAPiece()) {
96              return false;
97          }
98  
99          if (target.getPiece().getOwner() == owner) {
100             return false;
101         }
102 
103         return target.getPiece().getClass() == Pawn.class;
104     }
105 
106     /**
107      * Returns a list containing all squares this pawn can legally move to. That
108      * means squares diagonally forward (to pawn's owner's direction) of this
109      * pawn where is an opposing piece to be taken as well as square straight
110      * forward if it doesn't contain a piece. If it's pawn's first movement,
111      * then pawn can move up to two squares forward if there's no pieces of
112      * either owner on the way.
113      *
114      * @param board chessboard on which movement happens
115      * @return a list containing all squares this pawn can legally move to.
116      */
117     @Override
118     public Set<Square> possibleMoves(ChessBoardLogic board) {
119         Set<Square> moves = new HashSet<>();
120         int newrow = location.getrow() + owner.getDirection();
121 
122         addSquareIfWithinTableAndEmpty(board, newrow, moves);
123 
124         if (firstMovement()) {
125             newrow += owner.getDirection();
126             addSquareIfWithinTableAndEmpty(board, newrow, moves);
127         }
128 
129         addPossibilitiesToTakeOpposingPieces(board, moves);
130 
131         return moves;
132     }
133 
134     private void addPossibilitiesToTakeOpposingPieces(ChessBoardLogic board, Set<Square> moves) {
135         threatenedSquares(board).stream().filter(i -> legalToMoveTo(i, board))
136                 .filter(i -> i.containsAPiece())
137                 .forEach(i -> moves.add(i));
138     }
139 
140     private void addSquareIfWithinTableAndEmpty(ChessBoardLogic board, int newrow, Set<Square> moves) {
141         Square target;
142         if (board.withinTable(location.getcolumn(), newrow)) {
143             target = board.getSquare(location.getcolumn(), newrow);
144 
145             if (!target.containsAPiece()) {
146                 moves.add(target);
147             }
148         }
149     }
150 
151     private boolean firstMovement() {
152         if (owner == Player.WHITE) {
153             return location.getrow() == 1;
154         }
155         return location.getrow() == 6;
156     }
157 }
