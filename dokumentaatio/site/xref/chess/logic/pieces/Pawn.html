1   package chess.logic.pieces;
2   
3   import chess.logic.board.Player;
4   
5   /**
6    *
7    * This class corresponds physical bishop-piece thus only containing knowledge
8    * of it's place, whether or not is has moved yet and if it moved two squares
9    * last turn. Class also offers methods to access this information and clone
10   * this pawn.
11   *
12   * @author sami
13   */
14  public class Pawn extends Piece {
15  
16      private boolean movedTwoSquaresLastTurn;
17      private boolean hasBeenMoved;
18  
19      /**
20       * Creates a new Pawn-class object with given location and owner.
21       *
22       * @param column column of the square this pawn is placed on
23       * @param row row of the square this pawn is placed on
24       * @param owner owner of this pawn
25       */
26      public Pawn(int column, int row, Player owner) {
27          super(column, row, owner);
28          movedTwoSquaresLastTurn = false;
29          hasBeenMoved = false;
30      }
31  
32      public boolean getHasBeenMoved() {
33          return hasBeenMoved;
34      }
35  
36      public void setHasBeenMoved(boolean hasBeenMoved) {
37          this.hasBeenMoved = hasBeenMoved;
38      }
39  
40      public boolean getMovedTwoSquaresLastTurn() {
41          return movedTwoSquaresLastTurn;
42      }
43  
44      public void setMovedTwoSquaresLastTurn(boolean movedTwoSquaresLastTurn) {
45          this.movedTwoSquaresLastTurn = movedTwoSquaresLastTurn;
46      }
47  
48      @Override
49      public Piece clone() {
50          Pawn clone = new Pawn(column, row, owner);
51          clone.setMovedTwoSquaresLastTurn(movedTwoSquaresLastTurn);
52          return clone;
53      }
54  
55      @Override
56      public boolean equals(Object obj) {
57          if (!super.equals(obj)) {
58              return false;
59          }
60  
61          Pawn pawn = (Pawn) obj;
62  
63          if (hasBeenMoved != pawn.getHasBeenMoved()) {
64              return false;
65          }
66  
67          return movedTwoSquaresLastTurn == pawn.getMovedTwoSquaresLastTurn();
68      }
69  
70      /**
71       * Returns last row in the direction this pawn is moving towards.
72       *
73       * @return integer value of last row on board in direction of movement.
74       */
75      public int opposingEnd() {
76          if (owner == Player.BLACK) {
77              return 0;
78          }
79          return 7;
80      }
81  }
