1   package chess.pieces;

3   import chess.board.ChessBoard;
4   import java.util.ArrayList;
5   import java.util.List;
6   import chess.board.Player;
7   import chess.board.Square;
8   
9   public class Pawn extends Piece {
10  
11      public Pawn(Square square, Player owner) {
12          super(square, owner);
13      }
14  
15      @Override
16      public Piece clone(Square location) {
17          return new Pawn(location, this.owner);
18      }
19  
20      @Override
21      public List<Square> threatenedSquares(ChessBoard board) {
22          List<Square> squares = new ArrayList();
23          int[] fileChange = new int[]{1, -1};
24          int file = this.location.getFile();
25          int rank = this.location.getRank() + this.owner.getDirection();
26  
27          for (int i = 0; i < 2; i++) {
28              if (board.withinTable(file + fileChange[i], rank)) {
29                  squares.add(board.getSquare(file + fileChange[i], rank));
30              }
31          }
32  
33          return squares;
34      }
35  
36      @Override
37      public List<Square> possibleMoves(ChessBoard board) {
38          List<Square> moves = new ArrayList<>();
39          int newRank = location.getRank() + owner.getDirection();
40  
41          addSquareIfWithinTableAndEmpty(board, newRank, moves);
42  
43          if (firstMovement()) {
44              newRank += owner.getDirection();
45              addSquareIfWithinTableAndEmpty(board, newRank, moves);
46          }
47  
48          addPossibilitiesToTakeOpposingPieces(board, moves);
49  
50          return moves;
51      }
52  
53      private void addPossibilitiesToTakeOpposingPieces(ChessBoard board, List<Square> moves) {
54          threatenedSquares(board).stream().filter(i -> legalToMoveTo(i, board))
55                  .filter(i -> i.containsAPiece())
56                  .forEach(i -> moves.add(i));
57      }
58  
59      private void addSquareIfWithinTableAndEmpty(ChessBoard board, int newRank, List<Square> moves) {
60          Square target;
61          if (board.withinTable(location.getFile(), newRank)) {
62              target = board.getSquare(location.getFile(), newRank);
63  
64              if (!target.containsAPiece()) {
65                  moves.add(target);
66              }
67          }
68      }
69  
70      private boolean firstMovement() {
71          if (owner == Player.WHITE) {
72              return location.getRank() == 1;
73          }
74          return location.getRank() == 6;
75      }
76  }
